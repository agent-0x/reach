# Reach MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Go binary that lets AI agents remotely execute commands, read/write files on servers via HTTPS + Token auth, with MCP integration for Claude Code.

**Architecture:** Single Go binary with subcommands: `reach-agent` (server mode), `reach` (CLI client), `reach mcp serve` (MCP stdio server). Server listens HTTPS with self-signed TLS + Bearer Token. Client reads `~/.reach/config.yaml` for server configs. MCP server translates MCP tool calls to HTTP requests.

**Tech Stack:** Go 1.26, net/http (stdlib), crypto/tls, cobra (CLI), MCP Go SDK (mark3labs/mcp-go)

**Spec:** `docs/specs/2026-03-29-reach-design.md`

```go
// internal/agent/init.go
package agent

import (
	"crypto/ecdsa"
	"crypto/elliptic"
	"crypto/rand"
	"crypto/sha256"
	"crypto/x509"
	"crypto/x509/pkix"
	"encoding/hex"
	"encoding/pem"
	"fmt"
	"math/big"
	"net"
	"os"
	"path/filepath"
	"time"
)

type InitResult struct {
	Token       string
	Fingerprint string
	CertPath    string
	KeyPath     string
	ConfigPath  string
}

func Init(configDir string) (*InitResult, error) {
	if err := os.MkdirAll(configDir, 0700); err != nil {
		return nil, err
	}

	// Generate token
	tokenBytes := make([]byte, 64)
	if _, err := rand.Read(tokenBytes); err != nil {
		return nil, err
	}
	token := hex.EncodeToString(tokenBytes)

	// Generate TLS cert
	key, err := ecdsa.GenerateKey(elliptic.P256(), rand.Reader)
	if err != nil {
		return nil, err
	}

	serial, _ := rand.Int(rand.Reader, new(big.Int).Lsh(big.NewInt(1), 128))
	tmpl := &x509.Certificate{
		SerialNumber: serial,
		Subject:      pkix.Name{CommonName: "reach-agent"},
		NotBefore:    time.Now(),
		NotAfter:     time.Now().Add(10 * 365 * 24 * time.Hour),
		KeyUsage:     x509.KeyUsageDigitalSignature,
		ExtKeyUsage:  []x509.ExtKeyUsage{x509.ExtKeyUsageServerAuth},
	}

	// Add all local IPs as SANs
	addrs, _ := net.InterfaceAddrs()
	for _, a := range addrs {
		if ipnet, ok := a.(*net.IPNet); ok && !ipnet.IP.IsLoopback() {
			tmpl.IPAddresses = append(tmpl.IPAddresses, ipnet.IP)
		}
	}
	tmpl.IPAddresses = append(tmpl.IPAddresses, net.ParseIP("127.0.0.1"))

	certDER, err := x509.CreateCertificate(rand.Reader, tmpl, tmpl, &key.PublicKey, key)
	if err != nil {
		return nil, err
	}

	// Fingerprint
	fp := sha256.Sum256(certDER)
	fingerprint := "sha256:" + hex.EncodeToString(fp[:])

	// Write cert
	certPath := filepath.Join(configDir, "cert.pem")
	certFile, _ := os.Create(certPath)
	pem.Encode(certFile, &pem.Block{Type: "CERTIFICATE", Bytes: certDER})
	certFile.Close()

	// Write key
	keyPath := filepath.Join(configDir, "key.pem")
	keyFile, _ := os.OpenFile(keyPath, os.O_CREATE|os.O_WRONLY|os.O_TRUNC, 0600)
	keyDER, _ := x509.MarshalECPrivateKey(key)
	pem.Encode(keyFile, &pem.Block{Type: "EC PRIVATE KEY", Bytes: keyDER})
	keyFile.Close()

	// Write config
	configPath := filepath.Join(configDir, "config.yaml")
	configContent := fmt.Sprintf("port: 7100\ntoken: \"%s\"\ntls_cert: \"%s\"\ntls_key: \"%s\"\nmax_output: 10485760\nmax_timeout: 600\n",
		token, certPath, keyPath)
	os.WriteFile(configPath, []byte(configContent), 0600)

	return &InitResult{
		Token: token, Fingerprint: fingerprint,
		CertPath: certPath, KeyPath: keyPath, ConfigPath: configPath,
	}, nil
}
```

```go
// internal/agent/server.go
package agent

import (
	"encoding/json"
	"fmt"
	"log"
	"net/http"
	"os"
	"strings"

	"gopkg.in/yaml.v3"
)

type AgentConfig struct {
	Port       int    `yaml:"port"`
	Token      string `yaml:"token"`
	TLSCert    string `yaml:"tls_cert"`
	TLSKey     string `yaml:"tls_key"`
	MaxOutput  int    `yaml:"max_output"`
	MaxTimeout int    `yaml:"max_timeout"`
}

func LoadConfig(path string) (*AgentConfig, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, err
	}
	var cfg AgentConfig
	if err := yaml.Unmarshal(data, &cfg); err != nil {
		return nil, err
	}
	if cfg.Port == 0 {
		cfg.Port = 7100
	}
	if cfg.MaxOutput == 0 {
		cfg.MaxOutput = 10 * 1024 * 1024
	}
	if cfg.MaxTimeout == 0 {
		cfg.MaxTimeout = 600
	}
	return &cfg, nil
}

func Serve(cfg *AgentConfig) error {
	mux := http.NewServeMux()
	mux.HandleFunc("/health", handleHealth)
	mux.HandleFunc("/info", handleInfo)
	mux.HandleFunc("/exec", makeHandler(cfg, handleExec))
	mux.HandleFunc("/read", makeHandler(cfg, handleRead))
	mux.HandleFunc("/write", makeHandler(cfg, handleWrite))
	mux.HandleFunc("/upload", makeHandler(cfg, handleUpload))
	mux.HandleFunc("/download", makeHandler(cfg, handleDownload))

	addr := fmt.Sprintf(":%d", cfg.Port)
	log.Printf("reach-agent listening on https://0.0.0.0%s", addr)
	return http.ListenAndServeTLS(addr, cfg.TLSCert, cfg.TLSKey, authMiddleware(cfg.Token, mux))
}

func authMiddleware(token string, next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		if r.URL.Path == "/health" {
			next.ServeHTTP(w, r)
			return
		}
		auth := r.Header.Get("Authorization")
		if !strings.HasPrefix(auth, "Bearer ") || auth[7:] != token {
			http.Error(w, `{"error":"unauthorized"}`, http.StatusUnauthorized)
			return
		}
		next.ServeHTTP(w, r)
	})
}

type handler func(*AgentConfig, http.ResponseWriter, *http.Request)

func makeHandler(cfg *AgentConfig, fn handler) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		fn(cfg, w, r)
	}
}

func jsonResponse(w http.ResponseWriter, v interface{}) {
	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(v)
}

func jsonError(w http.ResponseWriter, msg string, code int) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(code)
	json.NewEncoder(w).Encode(map[string]string{"error": msg})
}
```

```go
// internal/agent/exec.go
package agent

import (
	"bytes"
	"context"
	"os/exec"
	"syscall"
	"time"
)

type ExecResult struct {
	Stdout    string `json:"stdout"`
	Stderr    string `json:"stderr"`
	ExitCode  int    `json:"exit_code"`
	Truncated bool   `json:"truncated"`
}

func Execute(command string, timeout int, maxOutput int) *ExecResult {
	ctx, cancel := context.WithTimeout(context.Background(), time.Duration(timeout)*time.Second)
	defer cancel()

	cmd := exec.CommandContext(ctx, "sh", "-c", command)
	cmd.SysProcAttr = &syscall.SysProcAttr{Setpgid: true}

	var stdout, stderr bytes.Buffer
	cmd.Stdout = &stdout
	cmd.Stderr = &stderr

	err := cmd.Run()

	// Kill process group on timeout
	if ctx.Err() == context.DeadlineExceeded {
		if cmd.Process != nil {
			syscall.Kill(-cmd.Process.Pid, syscall.SIGKILL)
		}
	}

	result := &ExecResult{
		Stdout: stdout.String(),
		Stderr: stderr.String(),
	}

	if err != nil {
		if exitErr, ok := err.(*exec.ExitError); ok {
			result.ExitCode = exitErr.ExitCode()
		} else if ctx.Err() == context.DeadlineExceeded {
			result.ExitCode = 124 // timeout convention
			result.Stderr += "\nreach: command timed out"
		} else {
			result.ExitCode = 1
			result.Stderr += "\n" + err.Error()
		}
	}

	// Truncate if too large
	if len(result.Stdout) > maxOutput {
		result.Stdout = result.Stdout[:maxOutput]
		result.Truncated = true
	}
	if len(result.Stderr) > maxOutput {
		result.Stderr = result.Stderr[:maxOutput]
		result.Truncated = true
	}

	return result
}
```

```go
// internal/agent/fileops.go
package agent

import (
	"fmt"
	"io/fs"
	"os"
	"path/filepath"
)

type ReadResult struct {
	Content   string `json:"content"`
	Size      int    `json:"size"`
	Truncated bool   `json:"truncated"`
}

func ReadFile(path string, maxSize int) (*ReadResult, error) {
	info, err := os.Stat(path)
	if err != nil {
		return nil, err
	}

	data, err := os.ReadFile(path)
	if err != nil {
		return nil, err
	}

	result := &ReadResult{
		Content: string(data),
		Size:    int(info.Size()),
	}

	if len(data) > maxSize {
		result.Content = string(data[:maxSize])
		result.Truncated = true
	}

	return result, nil
}

func WriteFile(path string, content string, mode string) error {
	// Determine file mode
	var fileMode fs.FileMode = 0644
	if mode != "" {
		fmt.Sscanf(mode, "%o", &fileMode)
	} else {
		// Preserve existing mode if file exists
		if info, err := os.Stat(path); err == nil {
			fileMode = info.Mode()
		}
	}

	// Atomic write: temp file → fsync → rename
	dir := filepath.Dir(path)
	tmp, err := os.CreateTemp(dir, ".reach-write-*")
	if err != nil {
		return err
	}
	tmpPath := tmp.Name()

	if _, err := tmp.WriteString(content); err != nil {
		tmp.Close()
		os.Remove(tmpPath)
		return err
	}
	if err := tmp.Sync(); err != nil {
		tmp.Close()
		os.Remove(tmpPath)
		return err
	}
	tmp.Close()

	if err := os.Chmod(tmpPath, fileMode); err != nil {
		os.Remove(tmpPath)
		return err
	}

	if err := os.Rename(tmpPath, path); err != nil {
		os.Remove(tmpPath)
		return err
	}

	return nil
}
```

```go
// internal/agent/handlers.go
package agent

import (
	"encoding/json"
	"io"
	"net/http"
	"os"
	"runtime"
	"time"

	"github.com/shirou/gopsutil/v4/cpu"
	"github.com/shirou/gopsutil/v4/disk"
	"github.com/shirou/gopsutil/v4/host"
	"github.com/shirou/gopsutil/v4/mem"
)

const Version = "0.1.0"

func handleHealth(w http.ResponseWriter, r *http.Request) {
	jsonResponse(w, map[string]interface{}{"ok": true, "version": Version})
}

func handleInfo(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	hostname, _ := os.Hostname()
	memInfo, _ := mem.VirtualMemory()
	diskInfo, _ := disk.Usage("/")
	hostInfo, _ := host.Info()
	cpuCount, _ := cpu.Counts(true)

	jsonResponse(w, map[string]interface{}{
		"hostname":      hostname,
		"os":            runtime.GOOS,
		"arch":          runtime.GOARCH,
		"cpu_count":     cpuCount,
		"mem_total_mb":  memInfo.Total / 1024 / 1024,
		"mem_free_mb":   memInfo.Available / 1024 / 1024,
		"disk_total_gb": diskInfo.Total / 1024 / 1024 / 1024,
		"disk_free_gb":  diskInfo.Free / 1024 / 1024 / 1024,
		"uptime_secs":   hostInfo.Uptime,
		"agent_version": Version,
	})
}

func handleExec(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		jsonError(w, "method not allowed", 405)
		return
	}
	var req struct {
		Command string `json:"command"`
		Timeout int    `json:"timeout"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		jsonError(w, "invalid json", 400)
		return
	}
	if req.Command == "" {
		jsonError(w, "command required", 400)
		return
	}
	if req.Timeout <= 0 {
		req.Timeout = 30
	}
	if req.Timeout > cfg.MaxTimeout {
		req.Timeout = cfg.MaxTimeout
	}

	result := Execute(req.Command, req.Timeout, cfg.MaxOutput)
	jsonResponse(w, result)
}

func handleRead(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		jsonError(w, "method not allowed", 405)
		return
	}
	var req struct {
		Path string `json:"path"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		jsonError(w, "invalid json", 400)
		return
	}
	if req.Path == "" {
		jsonError(w, "path required", 400)
		return
	}

	result, err := ReadFile(req.Path, cfg.MaxOutput)
	if err != nil {
		jsonError(w, err.Error(), 404)
		return
	}
	jsonResponse(w, result)
}

func handleWrite(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		jsonError(w, "method not allowed", 405)
		return
	}
	var req struct {
		Path    string `json:"path"`
		Content string `json:"content"`
		Mode    string `json:"mode"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		jsonError(w, "invalid json", 400)
		return
	}
	if req.Path == "" {
		jsonError(w, "path required", 400)
		return
	}

	if err := WriteFile(req.Path, req.Content, req.Mode); err != nil {
		jsonError(w, err.Error(), 500)
		return
	}
	jsonResponse(w, map[string]bool{"ok": true})
}

func handleUpload(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		jsonError(w, "method not allowed", 405)
		return
	}
	r.ParseMultipartForm(100 << 20) // 100MB max
	file, _, err := r.FormFile("file")
	if err != nil {
		jsonError(w, "file required", 400)
		return
	}
	defer file.Close()

	path := r.FormValue("path")
	if path == "" {
		jsonError(w, "path required", 400)
		return
	}

	data, err := io.ReadAll(file)
	if err != nil {
		jsonError(w, err.Error(), 500)
		return
	}

	if err := WriteFile(path, string(data), ""); err != nil {
		jsonError(w, err.Error(), 500)
		return
	}
	jsonResponse(w, map[string]interface{}{"ok": true, "size": len(data)})
}

func handleDownload(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	path := r.URL.Query().Get("path")
	if path == "" {
		jsonError(w, "path required", 400)
		return
	}

	f, err := os.Open(path)
	if err != nil {
		jsonError(w, err.Error(), 404)
		return
	}
	defer f.Close()

	info, _ := f.Stat()
	w.Header().Set("Content-Type", "application/octet-stream")
	w.Header().Set("Content-Disposition", "attachment; filename=\""+info.Name()+"\"")
	http.ServeContent(w, r, info.Name(), info.ModTime(), f)
}
```

```go
// internal/client/config.go
package client

import (
	"os"
	"path/filepath"

	"gopkg.in/yaml.v3"
)

type ServerConfig struct {
	Host        string `yaml:"host"`
	Port        int    `yaml:"port"`
	Token       string `yaml:"token"`
	Fingerprint string `yaml:"fingerprint"`
}

type Config struct {
	Servers map[string]*ServerConfig `yaml:"servers"`
}

func ConfigPath() string {
	home, _ := os.UserHomeDir()
	return filepath.Join(home, ".reach", "config.yaml")
}

func LoadClientConfig() (*Config, error) {
	path := ConfigPath()
	data, err := os.ReadFile(path)
	if err != nil {
		if os.IsNotExist(err) {
			return &Config{Servers: map[string]*ServerConfig{}}, nil
		}
		return nil, err
	}
	var cfg Config
	if err := yaml.Unmarshal(data, &cfg); err != nil {
		return nil, err
	}
	if cfg.Servers == nil {
		cfg.Servers = map[string]*ServerConfig{}
	}
	return &cfg, nil
}

func (c *Config) Save() error {
	path := ConfigPath()
	dir := filepath.Dir(path)
```

```go
// internal/client/client.go
package client

import (
	"bytes"
	"crypto/sha256"
	"crypto/tls"
	"encoding/hex"
	"encoding/json"
	"fmt"
	"io"
	"mime/multipart"
	"net/http"
	"os"
	"time"
)

type ReachClient struct {
	Name   string
	Config *ServerConfig
	http   *http.Client
}

func NewClient(name string, cfg *ServerConfig) *ReachClient {
	tlsConfig := &tls.Config{
		InsecureSkipVerify: true, // We verify fingerprint manually
		VerifyConnection: func(cs tls.ConnectionState) error {
			if cfg.Fingerprint == "" {
				return nil // TOFU: first connection, will be saved
			}
			if len(cs.PeerCertificates) == 0 {
				return fmt.Errorf("no server certificate")
			}
			fp := sha256.Sum256(cs.PeerCertificates[0].Raw)
			got := "sha256:" + hex.EncodeToString(fp[:])
			if got != cfg.Fingerprint {
				return fmt.Errorf("fingerprint mismatch: expected %s, got %s", cfg.Fingerprint, got)
			}
			return nil
		},
	}

	return &ReachClient{
		Name:   name,
		Config: cfg,
		http: &http.Client{
			Timeout:   60 * time.Second,
			Transport: &http.Transport{TLSClientConfig: tlsConfig},
		},
	}
}

func (c *ReachClient) baseURL() string {
	port := c.Config.Port
	if port == 0 {
		port = 7100
	}
	return fmt.Sprintf("https://%s:%d", c.Config.Host, port)
}

func (c *ReachClient) do(method, path string, body interface{}) ([]byte, error) {
	var bodyReader io.Reader
	if body != nil {
		data, _ := json.Marshal(body)
		bodyReader = bytes.NewReader(data)
	}

	req, err := http.NewRequest(method, c.baseURL()+path, bodyReader)
	if err != nil {
		return nil, err
	}
	req.Header.Set("Authorization", "Bearer "+c.Config.Token)
	if body != nil {
		req.Header.Set("Content-Type", "application/json")
	}

	resp, err := c.http.Do(req)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()

	data, err := io.ReadAll(resp.Body)
	if err != nil {
		return nil, err
	}

	if resp.StatusCode != 200 {
		return nil, fmt.Errorf("HTTP %d: %s", resp.StatusCode, string(data))
	}
	return data, nil
}

func (c *ReachClient) Exec(command string, timeout int) (map[string]interface{}, error) {
	data, err := c.do("POST", "/exec", map[string]interface{}{"command": command, "timeout": timeout})
	if err != nil {
		return nil, err
	}
	var result map[string]interface{}
	json.Unmarshal(data, &result)
	return result, nil
}

func (c *ReachClient) ReadFile(path string) (map[string]interface{}, error) {
	data, err := c.do("POST", "/read", map[string]string{"path": path})
	if err != nil {
		return nil, err
	}
	var result map[string]interface{}
	json.Unmarshal(data, &result)
	return result, nil
}

func (c *ReachClient) WriteFile(path, content string) error {
	_, err := c.do("POST", "/write", map[string]string{"path": path, "content": content})
	return err
}

func (c *ReachClient) Upload(localPath, remotePath string) (int64, error) {
	file, err := os.Open(localPath)
	if err != nil {
		return 0, err
	}
	defer file.Close()

	var buf bytes.Buffer
	writer := multipart.NewWriter(&buf)
	part, _ := writer.CreateFormFile("file", localPath)
	io.Copy(part, file)
	writer.WriteField("path", remotePath)
	writer.Close()

	req, _ := http.NewRequest("POST", c.baseURL()+"/upload", &buf)
	req.Header.Set("Authorization", "Bearer "+c.Config.Token)
	req.Header.Set("Content-Type", writer.FormDataContentType())

	resp, err := c.http.Do(req)
	if err != nil {
		return 0, err
	}
	defer resp.Body.Close()

	if resp.StatusCode != 200 {
		data, _ := io.ReadAll(resp.Body)
		return 0, fmt.Errorf("HTTP %d: %s", resp.StatusCode, string(data))
	}

	var result map[string]interface{}
	json.NewDecoder(resp.Body).Decode(&result)
	size, _ := result["size"].(float64)
	return int64(size), nil
}

func (c *ReachClient) Download(remotePath, localPath string) error {
	req, _ := http.NewRequest("GET", c.baseURL()+"/download?path="+remotePath, nil)
	req.Header.Set("Authorization", "Bearer "+c.Config.Token)

	resp, err := c.http.Do(req)
	if err != nil {
		return err
	}
	defer resp.Body.Close()

	if resp.StatusCode != 200 {
		data, _ := io.ReadAll(resp.Body)
		return fmt.Errorf("HTTP %d: %s", resp.StatusCode, string(data))
	}

	f, err := os.Create(localPath)
	if err != nil {
		return err
	}
	defer f.Close()
	_, err = io.Copy(f, resp.Body)
	return err
}

func (c *ReachClient) Info() (map[string]interface{}, error) {
	data, err := c.do("GET", "/info", nil)
	if err != nil {
		return nil, err
	}
	var result map[string]interface{}
	json.Unmarshal(data, &result)
	return result, nil
}

func (c *ReachClient) Health() (bool, error) {
	_, err := c.do("GET", "/health", nil)
	return err == nil, err
}

// GetFingerprint connects and returns the server cert fingerprint (for TOFU)
func (c *ReachClient) GetFingerprint() (string, error) {
	conn, err := tls.Dial("tcp", fmt.Sprintf("%s:%d", c.Config.Host, c.Config.Port), &tls.Config{InsecureSkipVerify: true})
	if err != nil {
		return "", err
	}
	defer conn.Close()

	certs := conn.ConnectionState().PeerCertificates
	if len(certs) == 0 {
		return "", fmt.Errorf("no certificates")
	}
	fp := sha256.Sum256(certs[0].Raw)
	return "sha256:" + hex.EncodeToString(fp[:]), nil
}
```
