package agent

import (
	"encoding/json"
	"net/http"
	"os"
	"runtime"
	"sort"
	"time"

	"github.com/shirou/gopsutil/v4/cpu"
	"github.com/shirou/gopsutil/v4/disk"
	"github.com/shirou/gopsutil/v4/host"
	"github.com/shirou/gopsutil/v4/load"
	"github.com/shirou/gopsutil/v4/mem"
	"github.com/shirou/gopsutil/v4/net"
	"github.com/shirou/gopsutil/v4/process"
)

type statsRequest struct {
	TopN int `json:"top_n"`
}

type cpuStats struct {
	Count        int       `json:"count"`
	UsagePercent float64   `json:"usage_percent"`
	PerCore      []float64 `json:"per_core"`
	LoadAvg      []float64 `json:"load_avg,omitempty"`
}

type memoryStats struct {
	TotalMB      uint64  `json:"total_mb"`
	UsedMB       uint64  `json:"used_mb"`
	AvailableMB  uint64  `json:"available_mb"`
	UsagePercent float64 `json:"usage_percent"`
	SwapTotalMB  uint64  `json:"swap_total_mb"`
	SwapUsedMB   uint64  `json:"swap_used_mb"`
}

type partitionStats struct {
	Mount        string  `json:"mount"`
	TotalGB      uint64  `json:"total_gb"`
	UsedGB       uint64  `json:"used_gb"`
	FreeGB       uint64  `json:"free_gb"`
	UsagePercent float64 `json:"usage_percent"`
}

type diskStats struct {
	Partitions []partitionStats `json:"partitions"`
}

type networkStats struct {
	BytesSent   uint64 `json:"bytes_sent"`
	BytesRecv   uint64 `json:"bytes_recv"`
	PacketsSent uint64 `json:"packets_sent"`
	PacketsRecv uint64 `json:"packets_recv"`
}

type processInfo struct {
	PID        int32   `json:"pid"`
	Name       string  `json:"name"`
	CPUPercent float64 `json:"cpu_percent"`
	MemoryMB   float64 `json:"memory_mb"`
}

func handleStats(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		jsonError(w, "method not allowed", http.StatusMethodNotAllowed)
		return
	}

	var req statsRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		// Empty body is OK, use defaults
		req = statsRequest{}
	}
	if req.TopN <= 0 {
		req.TopN = 5
	}
	if req.TopN > 20 {
		req.TopN = 20
	}

	result := map[string]interface{}{
		"agent_version": AgentVersion,
		"os":            runtime.GOOS,
		"arch":          runtime.GOARCH,
	}

	// Hostname
	hostname, _ := os.Hostname()
	result["hostname"] = hostname

	// Uptime
	if hostInfo, err := host.Info(); err == nil {
		result["uptime_secs"] = hostInfo.Uptime
	} else {
		result["uptime_secs"] = 0
	}

	// CPU
	result["cpu"] = collectCPU()

	// Memory
	result["memory"] = collectMemory()

	// Disk
	result["disk"] = collectDisk()

	// Network
	result["network"] = collectNetwork()

	// Top processes
	result["top_processes"] = collectTopProcesses(req.TopN)

	jsonResponse(w, result)
}

func collectCPU() cpuStats {
	stats := cpuStats{}

	if counts, err := cpu.Counts(true); err == nil {
		stats.Count = counts
	}

	if percents, err := cpu.Percent(200*time.Millisecond, false); err == nil && len(percents) > 0 {
		stats.UsagePercent = percents[0]
	}

	if perCore, err := cpu.Percent(0, true); err == nil {
		stats.PerCore = perCore
	}

	if avg, err := load.Avg(); err == nil && avg != nil {
		stats.LoadAvg = []float64{avg.Load1, avg.Load5, avg.Load15}
	}

	return stats
}

func collectMemory() memoryStats {
	stats := memoryStats{}

	if vm, err := mem.VirtualMemory(); err == nil {
		stats.TotalMB = vm.Total / 1024 / 1024
		stats.UsedMB = vm.Used / 1024 / 1024
		stats.AvailableMB = vm.Available / 1024 / 1024
		stats.UsagePercent = vm.UsedPercent
	}

	if swap, err := mem.SwapMemory(); err == nil {
		stats.SwapTotalMB = swap.Total / 1024 / 1024
		stats.SwapUsedMB = swap.Used / 1024 / 1024
	}

	return stats
}

func collectDisk() diskStats {
	stats := diskStats{Partitions: []partitionStats{}}

	parts, err := disk.Partitions(false)
	if err != nil {
		return stats
	}

	seen := map[string]bool{}
	for _, p := range parts {
		if seen[p.Mountpoint] {
			continue
		}
		seen[p.Mountpoint] = true

		usage, err := disk.Usage(p.Mountpoint)
		if err != nil {
			continue
		}

		stats.Partitions = append(stats.Partitions, partitionStats{
			Mount:        p.Mountpoint,
			TotalGB:      usage.Total / 1024 / 1024 / 1024,
			UsedGB:       usage.Used / 1024 / 1024 / 1024,
			FreeGB:       usage.Free / 1024 / 1024 / 1024,
			UsagePercent: usage.UsedPercent,
		})
	}

	return stats
}

func collectNetwork() networkStats {
	stats := networkStats{}

	if counters, err := net.IOCounters(false); err == nil && len(counters) > 0 {
		stats.BytesSent = counters[0].BytesSent
		stats.BytesRecv = counters[0].BytesRecv
		stats.PacketsSent = counters[0].PacketsSent
		stats.PacketsRecv = counters[0].PacketsRecv
	}

	return stats
}

func collectTopProcesses(topN int) []processInfo {
	procs, err := process.Processes()
	if err != nil {
		return []processInfo{}
	}

	type procWithCPU struct {
		info processInfo
		cpu  float64
	}

	var all []procWithCPU
	for _, p := range procs {
		name, err := p.Name()
		if err != nil {
			continue
		}
		cpuPct, _ := p.CPUPercent()
		memInfo, _ := p.MemoryInfo()
		var memMB float64
		if memInfo != nil {
			memMB = float64(memInfo.RSS) / 1024 / 1024
		}

		all = append(all, procWithCPU{
			info: processInfo{
				PID:        p.Pid,
				Name:       name,
				CPUPercent: cpuPct,
				MemoryMB:   memMB,
			},
			cpu: cpuPct,
		})
	}

	sort.Slice(all, func(i, j int) bool {
		return all[i].cpu > all[j].cpu
	})

	result := make([]processInfo, 0, topN)
	for i := 0; i < len(all) && i < topN; i++ {
		result = append(result, all[i].info)
	}
	return result
}

package agent

import (

	"os"

// DryRunResult is the response from /dryrun
type DryRunResult struct {
	Risk         string            `json:"risk"`
	Score        int               `json:"score"`
	Reasons      []string          `json:"reasons"`
	Affected     *AffectedInfo     `json:"affected,omitempty"`
	Suggestion   string            `json:"suggestion,omitempty"`
	WouldExecute bool              `json:"would_execute"`
	Confidence   string            `json:"confidence"`
}

// AffectedInfo describes the target of a file-affecting command
type AffectedInfo struct {
	Type   string `json:"type"`
	Path   string `json:"path"`
	Files  int    `json:"files,omitempty"`
	SizeMB int64  `json:"size_mb,omitempty"`
}

type dryRunRequest struct {
	Command string `json:"command"`
}

// riskRule maps a pattern to a score and reason
type riskRule struct {
	Pattern *regexp.Regexp
	Score   int
	Reason  string
}

var riskRules = []riskRule{
	{regexp.MustCompile(`\brm\s+.*-[a-zA-Z]*r[a-zA-Z]*f`), 85, "Recursive force delete (rm -rf)"},
	{regexp.MustCompile(`\brm\s+.*-[a-zA-Z]*f[a-zA-Z]*r`), 85, "Recursive force delete (rm -fr)"},
	{regexp.MustCompile(`\brm\s`), 45, "File deletion (rm)"},
	{regexp.MustCompile(`\bsystemctl\s+(restart|stop)\b`), 55, "Service restart/stop"},
	{regexp.MustCompile(`\bsystemctl\s+start\b`), 30, "Service start"},
	{regexp.MustCompile(`\bchmod\b`), 45, "Permission change (chmod)"},
	{regexp.MustCompile(`\bchown\b`), 45, "Ownership change (chown)"},

	{regexp.MustCompile(`\byum\s+(install|remove|erase)\b`), 50, "Package install/remove (yum)"},
	{regexp.MustCompile(`\bdnf\s+(install|remove|erase)\b`), 50, "Package install/remove (dnf)"},

}

var hasVariableExpansion = regexp.MustCompile(`\$[\w{]|` + "`")

func handleDryRun(cfg *AgentConfig, w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		jsonError(w, "method not allowed", http.StatusMethodNotAllowed)
		return
	}

	var req dryRunRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		jsonError(w, "invalid JSON: "+err.Error(), http.StatusBadRequest)
		return
	}
	if req.Command == "" {
		jsonError(w, "command is required", http.StatusBadRequest)
		return
	}

	result := scoreCommand(req.Command, &cfg.Security)
	jsonResponse(w, result)
}

func scoreCommand(command string, security *SecurityConfig) DryRunResult {
	result := DryRunResult{
		Risk:         "low",
		Score:        0,
		Reasons:      []string{},
		WouldExecute: true,
		Confidence:   "high",
	}

	// Check blacklist first
	blacklist := BuildBlacklist(security)
	if err := CheckDangerous(command, blacklist); err != nil {
		result.Risk = "blocked"
		result.Score = 100
		result.Reasons = append(result.Reasons, "Blocked by command blacklist")
		result.WouldExecute = false
		result.Confidence = "high"
		return result
	}

	// Determine confidence
	if hasVariableExpansion.MatchString(command) {
		result.Confidence = "low"
		result.Reasons = append(result.Reasons, "Contains unresolved shell expansion — risk may be higher than scored")
	} else if hasGlobbing.MatchString(command) {
		result.Confidence = "medium"
	}

	// Split command on shell operators and score each part
	parts := splitShellCommand(command)
	maxScore := 0

	for _, part := range parts {
		part = strings.TrimSpace(part)
		if part == "" {
			continue
		}

		// Check if it's a shell wrapper
		if m := shellWrapperPattern.FindStringSubmatch(part); m != nil {
			innerParts := splitShellCommand(m[2])
			parts = append(parts, innerParts...)
			continue
		}

		// Check read-only first
		isReadOnly := false
		for _, ro := range readOnlyPatterns {
			if ro.MatchString(part) {
				isReadOnly = true
				break
			}
		}
		if isReadOnly {
			continue
		}

		// Score against risk rules
		for _, rule := range riskRules {
			if rule.Pattern.MatchString(part) {
				if rule.Score > maxScore {
					maxScore = rule.Score
				}
				result.Reasons = append(result.Reasons, rule.Reason)
			}
		}
	}

	// Apply minimum score for low-confidence commands
	if result.Confidence == "low" && maxScore < 50 {
		maxScore = 50
	}

	// If no rules matched and not read-only, it's an unknown command
	if maxScore == 0 && len(result.Reasons) == 0 {
		// Check if the whole command is read-only
		isReadOnly := false
		for _, ro := range readOnlyPatterns {
			if ro.MatchString(strings.TrimSpace(command)) {
				isReadOnly = true
				break
			}
		}
		if isReadOnly {
			maxScore = 5
			result.Reasons = append(result.Reasons, "Read-only command")
		} else {
			maxScore = 30
			result.Reasons = append(result.Reasons, "Unknown command — could not determine risk")
			if result.Confidence == "high" {
				result.Confidence = "medium"
			}
		}
	}

	result.Score = maxScore
	result.Risk = scoreToRisk(maxScore)

	// Try to extract affected path for file commands
	result.Affected = extractAffected(command)

	// Suggest safer alternative
	result.Suggestion = suggestAlternative(command, result.Risk)

	return result
}

func splitShellCommand(cmd string) []string {
	// Split on &&, ||, ;, | (but not ||)
	var parts []string
	current := strings.Builder{}
	i := 0
	for i < len(cmd) {
		if cmd[i] == '&' && i+1 < len(cmd) && cmd[i+1] == '&' {
			parts = append(parts, current.String())
			current.Reset()
			i += 2
		} else if cmd[i] == '|' && i+1 < len(cmd) && cmd[i+1] == '|' {
			parts = append(parts, current.String())
			current.Reset()
			i += 2
		} else if cmd[i] == '|' {
			parts = append(parts, current.String())
			current.Reset()
			i++
		} else if cmd[i] == ';' {
			parts = append(parts, current.String())
			current.Reset()
			i++
		} else {
			current.WriteByte(cmd[i])
			i++
		}
	}
	if current.Len() > 0 {
		parts = append(parts, current.String())
	}
	return parts
}

func scoreToRisk(score int) string {
	switch {
	case score >= 100:
		return "blocked"
	case score >= 70:
		return "high"
	case score >= 30:
		return "medium"
	default:
		return "low"
	}
}

func extractAffected(command string) *AffectedInfo {
	// Try to extract path from rm, mv, cp commands
	patterns := []*regexp.Regexp{
		regexp.MustCompile(`\brm\s+(?:-[a-zA-Z]+\s+)*(.+?)(?:\s*[;&|]|$)`),
		regexp.MustCompile(`\bmv\s+(?:-[a-zA-Z]+\s+)*(.+?)\s+`),
	}

	for _, p := range patterns {
		m := p.FindStringSubmatch(command)
		if m == nil || len(m) < 2 {
			continue
		}

		path := strings.TrimSpace(m[1])
		// Resolve symlinks
		resolved, err := filepath.EvalSymlinks(path)
		if err == nil {
			path = resolved
		}

		info, err := os.Stat(path)
		if err != nil {
			return &AffectedInfo{Type: "unknown", Path: path}
		}

		affected := &AffectedInfo{Path: path}
		if info.IsDir() {
			affected.Type = "directory"
			affected.Files = countFiles(path)
			affected.SizeMB = dirSizeMB(path)
		} else {
			affected.Type = "file"
			affected.Files = 1
			affected.SizeMB = info.Size() / 1024 / 1024
		}
		return affected
	}
	return nil
}

func countFiles(dir string) int {
	count := 0
	_ = filepath.Walk(dir, func(_ string, _ os.FileInfo, _ error) error {
		count++
		if count > 10000 {
			return filepath.SkipAll
		}
		return nil
	})
	return count
}

func dirSizeMB(dir string) int64 {
	var size int64
	count := 0
	_ = filepath.Walk(dir, func(_ string, info os.FileInfo, _ error) error {
		if info != nil && !info.IsDir() {
			size += info.Size()
		}
		count++
		if count > 10000 {
			return filepath.SkipAll
		}
		return nil
	})
	return size / 1024 / 1024
}

func suggestAlternative(command, risk string) string {
	if risk == "low" || risk == "blocked" {
		return ""
	}
	cmd := strings.TrimSpace(command)
	if strings.Contains(cmd, "rm -rf") || strings.Contains(cmd, "rm -fr") {
		return "Consider listing contents first: ls -la <path>"
	}
	if strings.Contains(cmd, "systemctl restart") || strings.Contains(cmd, "systemctl stop") {
		return "Consider checking status first: systemctl status <service>"
	}
	return ""
}

		mcp.WithString("server",

func handleReachStats(_ context.Context, req mcp.CallToolRequest) (*mcp.CallToolResult, error) {
	args := req.GetArguments()

	c, err := getClientFromArgs(args)
	if err != nil {
		return mcp.NewToolResultError(err.Error()), nil
	}

	topN := 5
	if t, ok := args["top_n"].(float64); ok && t > 0 {
		topN = int(t)
	}

	result, err := c.Stats(topN)
	if err != nil {
		return mcp.NewToolResultError(fmt.Sprintf("stats failed: %v", err)), nil
	}

	return mcp.NewToolResultText(toJSON(result)), nil
}

func handleReachDryRun(_ context.Context, req mcp.CallToolRequest) (*mcp.CallToolResult, error) {
	args := req.GetArguments()

	c, err := getClientFromArgs(args)
	if err != nil {
		return mcp.NewToolResultError(err.Error()), nil
	}

	command, _ := args["command"].(string)
	if command == "" {
		return mcp.NewToolResultError("parameter 'command' is required"), nil
	}

	result, err := c.DryRun(command)
	if err != nil {
		return mcp.NewToolResultError(fmt.Sprintf("dryrun failed: %v", err)), nil
	}

	return mcp.NewToolResultText(toJSON(result)), nil
}
